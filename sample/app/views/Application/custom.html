#{extends 'main.html' /} #{set title:'Profiling custom code' /} Profiling custom code is easy. Here's a sample: 
import play.modules.profiler.MiniProfiler;
import play.modules.profiler.Step;
...

    public static void custom() {
        step = MiniProfiler.step("database", "findAll");
        List result;
        try {
            result = Data.findAll();
        } finally {
            step.close();
        }
        render(result);
    }
